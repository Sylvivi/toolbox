        function findKeyCard(el) { while (el) { if (el.classList && el.classList.contains('card') && el.dataset.kid) return el; el = el.parentElement; } return null; }
        function findKeyHandle(el) { while (el) { if (el.classList && el.classList.contains('key-drag')) return el; el = el.parentElement; } return null; }

        function onKeyDS(e) {
            var t = e.touches ? e.touches[0] : e;
            var h = findKeyHandle(e.target); if (!h) return;
            var card = findKeyCard(h); if (!card) return;
            e.preventDefault();
            var r = card.getBoundingClientRect();
            var ghost = card.cloneNode(true);
            ghost.id = 'key-drag-ghost';
            ghost.style.cssText = 'position:fixed;left:'+r.left+'px;top:'+r.top+'px;width:'+r.width+'px;height:'+r.height+'px;z-index:999;pointer-events:none;opacity:0.85;transform:scale(1.03);transition:none;box-shadow:0 16px 48px rgba(0,0,0,0.18);border:2px solid #5aab8b;border-radius:16px;overflow:hidden;';
            document.body.appendChild(ghost);
            card.style.opacity = '0.25'; card.style.border = '2px dashed #5aab8b';
            keyDragState = { id: card.dataset.kid, card: card, ghost: ghost, offsetX: t.clientX - r.left, offsetY: t.clientY - r.top };
        }

        function onKeyDM(e) {
            if (!keyDragState) return; e.preventDefault();
            var t = e.touches ? e.touches[0] : e;
            keyDragState.ghost.style.left = (t.clientX - keyDragState.offsetX) + 'px';
            keyDragState.ghost.style.top = (t.clientY - keyDragState.offsetY) + 'px';
            document.querySelectorAll('#keyCardList .card').forEach(function(c) { c.classList.remove('drag-over-top', 'drag-over-bottom'); });
            keyDragState.ghost.style.display = 'none';
            var target = document.elementFromPoint(t.clientX, t.clientY);
            keyDragState.ghost.style.display = '';
            var tc = findKeyCard(target);
            if (tc && tc !== keyDragState.card) {
                var r = tc.getBoundingClientRect();
                if (t.clientY < r.top + r.height / 2) tc.classList.add('drag-over-top');
                else tc.classList.add('drag-over-bottom');
            }
            if (t.clientY < 60) window.scrollBy(0, -8);
            else if (t.clientY > window.innerHeight - 60) window.scrollBy(0, 8);
        }

        function onKeyDE(e) {
            if (!keyDragState) return;
            if (keyDragState.ghost.parentNode) keyDragState.ghost.parentNode.removeChild(keyDragState.ghost);
            keyDragState.card.style.opacity = ''; keyDragState.card.style.border = '';
            var tc = null, before = true;
            document.querySelectorAll('#keyCardList .card').forEach(function(c) {
                if (c.classList.contains('drag-over-top')) { tc = c; before = true; }
                if (c.classList.contains('drag-over-bottom')) { tc = c; before = false; }
                c.classList.remove('drag-over-top', 'drag-over-bottom');
            });
            if (tc && tc.dataset.kid !== keyDragState.id) {
                var data = getKeyData(), fi = -1;
                for (var i = 0; i < data.length; i++) { if (data[i].id === keyDragState.id) fi = i; }
                if (fi >= 0) {
                    var item = data.splice(fi, 1)[0];
                    var ni = -1;
                    for (var i = 0; i < data.length; i++) { if (data[i].id === tc.dataset.kid) { ni = i; break; } }
                    if (!before) ni++;
                    data.splice(ni, 0, item);
                    setKeyData(data); renderKeyCards();
                }
            }
            keyDragState = null;
        }

        function toggleCodeSearch(cardId) {
            var bar = document.querySelector('.card[data-sid="' + cardId + '"] .code-search-bar');

        function doCodeSearch(cardId, input) {
            var keyword = input.value;
            var card = document.querySelector('.card[data-sid="' + cardId + '"]');
            var pre = card.querySelector('pre');
            var info = card.querySelector('.code-search-info');
            var data = getSnipData();
            var item = null;
            for (var i = 0; i < data.length; i++) { if (data[i].id === cardId) { item = data[i]; break; } }
            if (!item) return;
            if (!keyword) { pre.innerHTML = esc(item.code); info.textContent = ''; delete codeSearchState[cardId]; return; }
            var escaped = esc(item.code), kw = esc(keyword), kwLower = kw.toLowerCase();
            var count = 0, result = '', lower = escaped.toLowerCase(), pos = 0;
            while (pos < escaped.length) {
                var idx = lower.indexOf(kwLower, pos);
                if (idx === -1) { result += escaped.slice(pos); break; }
                result += escaped.slice(pos, idx) + '<mark class="code-hl" data-mi="' + count + '">' + escaped.slice(idx, idx + kw.length) + '</mark>';
                count++; pos = idx + kw.length;
            }
            pre.innerHTML = result;
            codeSearchState[cardId] = { current: 0, total: count };
            if (count > 0) { info.textContent = '1 / ' + count; highlightCurrent(cardId); }
            else info.textContent = '无匹配';
        }

        function highlightCurrent(cardId) {
            var card = document.querySelector('.card[data-sid="' + cardId + '"]');
            if (!card) return;
            var marks = card.querySelectorAll('mark.code-hl');
            var state = codeSearchState[cardId];
            if (!state || marks.length === 0) return;
            marks.forEach(function(m) { m.classList.remove('current'); });
            var cur = marks[state.current];
            cur.classList.add('current');
            cur.scrollIntoView({ behavior: 'smooth', block: 'center' });
        }

        function doEditorSearch() {
            var keyword = document.getElementById('editorSearchInput').value;
            var ta = document.getElementById('snipFormCode');
            var info = document.getElementById('editorSearchInfo');
            if (!keyword) { info.textContent = ''; editorSearchState = { matches: [], current: -1 }; return; }
            var text = ta.value.toLowerCase(), kw = keyword.toLowerCase();
            var matches = [], pos = 0;
            while (pos < text.length) { var idx = text.indexOf(kw, pos); if (idx === -1) break; matches.push(idx); pos = idx + 1; }
            editorSearchState.matches = matches;
            if (matches.length > 0) { editorSearchState.current = 0; info.textContent = '1 / ' + matches.length; selectEditorMatch(); }
            else { editorSearchState.current = -1; info.textContent = '无匹配'; }
        }

        function jumpEditorMatch(dir) {

        function copyText(text, btn) {
            if (navigator.clipboard) {
                navigator.clipboard.writeText(text).then(function() { onCopyOk(btn); }).catch(function() { fallbackCopy(text, btn); });
            } else { fallbackCopy(text, btn); }
        }
        function fallbackCopy(text, btn) {
            var ta = document.createElement('textarea');
            ta.value = text; ta.style.position = 'fixed'; ta.style.left = '-9999px';
            document.body.appendChild(ta); ta.select(); document.execCommand('copy');
            document.body.removeChild(ta); onCopyOk(btn);
        }
